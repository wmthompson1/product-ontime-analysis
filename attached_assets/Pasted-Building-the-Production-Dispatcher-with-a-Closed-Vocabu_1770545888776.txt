The LLM isn't writing SQL; it’s acting as a **Semantic Router** for the `SolderEngine`.

#### 1. The Dispatcher Logic (Closed-Vocabulary)

This function sits between the User and the Solder.

```python
import openai
import os

class ProductionDispatcher:
    def __init__(self, solder_engine, sqlite_conn):
        self.solder = solder_engine
        self.conn = sqlite_conn
        # Fetching current vocabulary from your DB
        self.intents = [r[0] for r in self.conn.execute("SELECT intent_name FROM schema_intents").fetchall()]
        self.concepts = [r[0] for r in self.conn.execute("SELECT concept_name FROM schema_concepts").fetchall()]

    def extract_semantic_intent(self, user_query):
        """
        Forces the LLM to return a JSON object mapping to our exact schema.
        """
        prompt = f"""
        Map the user's question to exactly one Intent and a list of Concepts.
        AVAILABLE INTENTS: {', '.join(self.intents)}
        AVAILABLE CONCEPTS: {', '.join(self.concepts)}

        USER QUESTION: "{user_query}"
        
        RETURN JSON ONLY: {{"intent": "...", "concepts": ["...", "..."], "perspective": "..."}}
        """
        
        # Using OpenAI to route (or Mock for Demo Mode)
        response = openai.ChatCompletion.create(
            model="gpt-4-turbo-preview",
            messages=[{"role": "system", "content": "You are a semantic router for a manufacturing graph."},
                      {"role": "user", "content": prompt}],
            response_format={ "type": "json_object" }
        )

    def dispatch(self, user_query, perspective_override=None):
        # 1. Routing
        semantic_map = self.extract_semantic_intent(user_query)
        
        # 2. Perspectival Adjustment
        perspective = perspective_override or semantic_map.get("perspective", "Shop_Floor")
        
        # 3. Soldering
        final_sql = self.solder.assemble_query(
            intent=semantic_map["intent"],
            perspective=perspective,
            concepts=semantic_map["concepts"],
            base_table="stg_manufacturing_flat"
        )
        
        return {
            "sql": final_sql,
            "intent": semantic_map["intent"],
            "concepts": semantic_map["concepts"],
            "perspective": perspective
        }
```
